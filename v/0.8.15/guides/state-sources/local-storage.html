Local Storage State Source

The Local Storage State Source allows you to persist data to localStorage.

var FooStorage = Marty.createStateSource({
  type: 'localStorage',
  saveFoo: function (foo) {
    this.set('foo', foo);
  },
  getFoo: function () {
    return this.get('foo');
  }
});

FooStorage.getFoo();
