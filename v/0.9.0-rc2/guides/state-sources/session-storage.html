Session Storage State Source

The Session Storage State Source allows you to persist data to sessionStorage.

var FooStorage = Marty.createStateSource({
  id: 'FooStorage',
  type: 'sessionStorage',
  saveFoo: function (foo) {
    this.set('foo', foo);
  },
  getFoo: function () {
    return this.get('foo');
  }
});

FooStorage.saveFoo('Foo');
class FooStorage extends Marty.LocalStorageStateSource {
  saveFoo(foo) {
    this.set('foo', foo);
  }
  getFoo() {
    return this.get('foo');
  }
}

FooStorage.saveFoo('Foo');
